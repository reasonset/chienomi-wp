IC recorderデータのpick upスクリプト

単純に日付でディレクトリをつくり、そこに移動するだけ。mvよりcp+rmのほうが細かく制御することが可能なので分けている。

#!/usr/bin/zsh target="$HOME/local/usr/media/sound/voice/record/$(date +%y%m%d)" if [[ ! -e $target ]] then mkdir -p $target fi cp **/*.(mp3|MP3|wav|WAV|aiff|ogg) $target && rm **/*.(mp3|MP3|wav|WAV|aiff|ogg)
